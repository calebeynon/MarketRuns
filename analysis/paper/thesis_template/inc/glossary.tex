\newglossaryentry{pi}
{
  name={\ensuremath{\pi(t)}},
  description={Bayesian posterior probability that the state of the world is good at time $t$},
  sort=pi
}

\newglossaryentry{lambda}
{
  name={\ensuremath{\lambda}},
  description={Probability of market termination in each period (0.125)},
  sort=lambda
}

\newglossaryentry{muB}
{
  name={\ensuremath{\mu_B}},
  description={Probability of a bad signal given the bad state (0.675)},
  sort=muB
}

\newglossaryentry{muG}
{
  name={\ensuremath{\mu_G}},
  description={Probability of a good signal given the bad state (0.325)},
  sort=muG
}
